    function checkproperty(){
        function Car(make, year){
            this.make = make;
            this.year = year;
        }

        let car1 = new Car("Mazda", "2017");
        let car2 = new Car("Honda", "2019");

        //checking for existing property
        result1 = car1.hasOwnProperty("make");
        //should be true
        result2 = car2.hasOwnProperty("model");
        //should be false
        document.querySelector("#hasProperty").innerHTML = result1;
        document.querySelector("#notOwnProperty").innerHTML = result2;
